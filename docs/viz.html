Gráfico para responder la pregunta 1

Vamos a hacer un

alt.Chart(penguins).mark_rect().encode(
    x='species:N',
    y='sex:N',
    color='mean(body_mass_g):Q'
)
alt.Chart(penguins).mark_bar().encode(
    x=alt.X('body_mass_g:Q', bin=True),
    y='count()',
    color='species:N'
)

Pregunta 3

alt.Chart(penguins).mark_boxplot().encode(
    x='sex:N',
    y='body_mass_g:Q',
    color='sex:N'
)
alt.Chart(penguins).mark_boxplot(size=60).encode(
    x=alt.X('sex:N', title='Sexo'),
    y=alt.Y('body_mass_g:Q', title='Masa corporal (g)'),
    color=alt.Color('sex:N', legend=None)
).properties(
    title='Comparación de masa corporal entre machos y hembras'
)
